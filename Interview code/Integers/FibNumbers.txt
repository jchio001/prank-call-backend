import java.io.*;
import java.util.*;

/*
 * To execute Java, please define "static void main" on a class
 * named Solution.
 *
 * If you need more classes, simply define them inline.
 */
class Solution {
  
  
  public static int normalFib(int num) {
    int fibArr[];
    if (num >= 0)
      fibArr = new int[num + 1];
    else
      return -1;
      
    fibArr[0] = 0;
    fibArr[1] = 1;
    int length = fibArr.length;
    for (int i = 2; i < fibArr.length; ++i) {
      fibArr[i] = fibArr[i -1] + fibArr[i - 2];
    }
    return fibArr[num];    
  }
  
  public static int recvFib(int num) {
    if (num <= 0)
      return 0;
    if (num == 1)
      return 1;
    
    return recvFib(num - 1) + recvFib(num - 2);    
  }
  
  public static void main(String[] args) {    
    System.out.println(normalFib(5));
  }
}
