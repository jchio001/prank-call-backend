import java.io.*;
import java.util.*;

/*
 * To execute Java, please define "static void main" on a class
 * named Solution.
 *
 * If you need more classes, simply define them inline.
 */

class Solution {    
  
  public static int binSearch(int arr[], int start, int end, int val) {        
        
    if (start > end)
      return -1;
    
    int middle = (start + end)/2;
    if (arr[middle] == val)
      return middle;
    if (start == end)
      return -1;
    
    if (val > arr[middle])
      return binSearch(arr, middle + 1, end, val);
    else
      return binSearch(arr, start, middle, val);
        
  }
      
  public static void main(String[] args) {    
    int[] myArr = {1, 2, 3, 4, 5, 6, 7, 8};
    System.out.println(binSearch(myArr, 0, 7, 4));
  }
}
