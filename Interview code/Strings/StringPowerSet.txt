import java.io.*;
import java.util.*;

/*

 * named Solution.

 * If you need more classes, simply define them inline.
 */

class Solution {    
    
  public static void main(String[] args) {    
    String strArr[] = {"All", "Hail", "King", "Aravind"};
    ArrayList< ArrayList<String>> powSet = new ArrayList < ArrayList<String>>();
    int width = strArr.length;
    int height = (int) Math.pow(2, width);        
    int interval = height;
            
    ArrayList<String> subset;    
    int value;
    for (int i = 0; i < height; ++i) {
      subset = new ArrayList<String>();
      for (int j = 0; j < width; ++j) {                        
        //I don't think I can explain what I'm doing with value in a few comments
        //Basically I'm generating values for a truth table
        //If I get a value of 1 (true) I add strArr[j] to the set
        value = (((int) Math.pow(2, j + 1)) * i/interval) % 2;        
        if (value == 1)
          subset.add(strArr[j]);
      }                  
      powSet.add(subset);
    }
    
    for (ArrayList<String> tmp : powSet) {
      for (String tmpStr : tmp) {
        System.out.print(tmpStr + " ");
      }
      System.out.println();
    }
    
  }
}
